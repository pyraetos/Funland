#version 330

layout (location=0) in vec3 position;
layout (location=1) in vec3 normal;

out vec3 viewedVertex;
out vec3 fragNormal;

uniform mat4 proj;
uniform mat4 view;
uniform mat4 model;

void main()
{
	vec4 viewedVertex4 = view * model * vec4(position, 1.0);
	vec4 vertex4 = proj * viewedVertex4;
	viewedVertex = viewedVertex4.xyz;
    gl_Position = vertex4;
    
    fragNormal = normalize(view * model * vec4(normal, 0.0)).xyz;
}
