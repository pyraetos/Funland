layout (location=0) in vec3 position;

out vec3 localVertex;

uniform mat4 proj;
uniform mat4 view;
uniform mat4 model;

void main()
{
	vec4 vertex4 = (proj * view * model) * vec4(position, 1.0);
    gl_Position = vertex4;
    localVertex = position;

}
