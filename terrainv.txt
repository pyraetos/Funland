#version 330

layout (location=0) in vec3 position;
layout (location=1) in float y;
layout (location=2) in vec3 inNormal;

out vec3 fragNormal;
out vec3 viewedVertex;

uniform mat4 proj;
uniform mat4 view;
uniform mat4 model;

void main()
{
	vec3 positionWithY = vec3(position.x, y, position.z);
	vec4 viewedVertex4 = view * model * vec4(positionWithY, 1.0);
	vec4 vertex4 = proj * viewedVertex4;
	viewedVertex = viewedVertex4.xyz;
    gl_Position = vertex4;
    fragNormal = inNormal;
}
